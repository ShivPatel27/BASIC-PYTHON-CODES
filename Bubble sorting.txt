n=int(input("Enter the length of the array: "))
A=[]
print('Give ',n,' no of values')
for i in range (n):
   A.append(int(input()))
   
z=1

for i in range  (n-1):
    c=0
    
    for j in range (n-z):
        if A[j] < A[j+1]:
            d=A[j+1]
            A[j+1]=A[j]
            A[j]=d
            c=1
    z=z+1
    print(A)
    print(c)
    if c==0:
        print('end')
        break
    
print('the array in decending order',A)
