n=int(input("Enter the length of array: "))
a=[]
print("Enter the values of array: ")
for i in range(n):
    a.append(int(input()))
b=[]
c=[]
for i in range(n):
    if a[i]>0:
        b.append(a[i])
    else:
        c.append(a[i])
        
print("The array with positive numbers is ",b)
print("The array with negative numbers is ",c)
