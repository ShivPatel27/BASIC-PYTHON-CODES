n=int(input("Enter value of n greater than 1: "))
a=[]
b=[]
for i in range(n):
    a.append("any number")
a[0]=0
a[1]=1

for i in range(2,n):
    a[i]=a[i-1]+a[i-2] 
    
for j in range(2,n):
    b.append(a[j]/a[j-1])
print("The fabonacci series: \n",a,"\n The ratio of adjacent no. :\n",b)
