q1=float(input("Enter a charge one"))
q2=float(input("Enter a charge two"))
r=float(input("Enter the distance between 2 charges"))
k=9*10**9
f=k*(q1*q2/r**2)
print("The Electric Force between your charges is: ",f)
e=k*(q1/r**2)
print("The Electric field of your first charge is: ",e)
e=k*(q2/r**2)
print("The Electric field of your second charge is: ",e)
p=k*(q1/r)
print("Potential at your first charge is: ",p)
p=k*(q2/r)
print("Potential at your second charge is: ",p)
